│                │                                                             │
│                ▼                                                             │
│   FilterContext state updates                                               │
│                │                                                             │

│                ├─► cameraSettings: { iso: { min: 100, max: 800 } }          │
│                └─► notes: { hasNotes: true }                                │
│                │                                                             │
│                ▼                                                             │
│   getActiveFilterSummaries() processes filter state                         │
│                │                                                             │
│                ▼                                                             │
│   Returns array of summaries:                                               │
│   [                                                                          │
│     { type: 'dateRange', label: 'Date', value: 'Last 7 Days' },             │
│     { type: 'tags', label: 'Tags', value: '2 (any)' },                      │
│     { type: 'species', label: 'Species', value: '1 selected' },             │
│     { type: 'cameraSettings', label: 'ISO', value: '100 - 800' },           │
│     { type: 'notes', label: 'Notes', value: 'Has notes' }                   │
│   ]                                                                          │
│                │                                                             │
│                ▼                                                             │
│   ActiveFilterChips renders chips for each summary                          │
│                │                                                             │
│                ▼                                                             │
│   User clicks remove (×) button                                             │
│                │                                                             │
│                ▼                                                             │
│   clearFilter(type) called                                                  │
│                │                                                             │
│                ▼                                                             │
│   FilterContext state resets that filter                                    │
│                │                                                             │
│                ▼                                                             │
│   Component re-renders without that chip                                    │
